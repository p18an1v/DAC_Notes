Solution:


#include<fstream>
#include<iostream>
using namespace std;
class Student
{
private:
	char prnno[10];
	char name[20];
	char stream[20];
public:
	Student()
	{
	}
	Student(const char* a, const char* b, const char* c)
	{
		strcpy_s(prnno,strlen(a)+1, a);
		strcpy_s(name, strlen(b) + 1, b);
		strcpy_s(stream, strlen(c) + 1,c);
	}
	char* getPrnno()
	{
		return prnno;
	}
	char* getName()
	{
		return name;
	}
	char* getStream()
	{
		return stream;
	}
	friend ostream& operator<<(ostream&, Student&);
};
ostream& operator<<(ostream& ref1, Student& s2)
{
	ref1 << s2.prnno << "\t" << s2.name << "\t" << s2.stream << endl;
	return ref1;
}
fstream f("c:\\temp\\myfile.txt", ios::in | ios::out | ios::trunc);
void show(char* ptr)
{
	Student temp;
	f.seekg(0, ios::beg);
	while (f.read((char*)&temp, sizeof(temp)))
	{
		if (!strcmp(temp.getPrnno(), ptr))
		{
			cout << temp << endl;
			break;
		}
	}
}
int main()
{
	Student arr[] = { Student("p001","abc","btech"),Student("p002","xyz","be"),Student("p003","pqr","mtech") };
	f.write((char*)&arr, sizeof(arr));

}
