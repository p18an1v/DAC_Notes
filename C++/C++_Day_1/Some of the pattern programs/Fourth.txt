#include<iostream>
using namespace std;
int main()
{
	char ch = 'A';
	for (int i=0;i<6;i++)
	{
		for (int space = 0; space <= i; space++)
		{
			cout << " ";
		}
		for (int j = 6; j >i; j--)
		{
			cout << ch << " ";
			ch++;
		}
		cout << endl;
		ch = 'A';
	}
}
