#include<iostream>
#include<thread>
#include<mutex>
#include<windows.h>
using namespace std;

mutex m;

void myfun()
{
	lock_guard<mutex> lg(m);
	for (int i = 0; i < 10; i++)
	{
		cout << "Hello\t" << i << "\t by\t" << this_thread::get_id() << endl;
		Sleep(100);
	}
}
int main()
{
	thread t1(myfun);
	thread t2(myfun);
	t1.join();  
	t2.join();
	cout << "inside main thread"<<endl;
}
