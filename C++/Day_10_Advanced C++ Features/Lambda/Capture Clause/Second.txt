// // Lambda Closure by reference

#include<iostream>
using namespace std;


int main()
{
	int a = 10;
	// auto mymethod = []() {cout << a << endl; }; error for "a"
	auto mymethod = [&a]() {a = 25; };
	cout << "Value of a before lambda call is\t" << a << endl;
	mymethod();
	cout << "Value of a after lambda call is\t" << a << endl;

	//auto mymethod = [a]() {a = 25; };// expression must be lvalue
	
}
