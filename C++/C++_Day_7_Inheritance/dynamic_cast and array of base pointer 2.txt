#include<iostream>
using namespace std;

class base
{
public:
	virtual void disp()
	{
		cout<<"in base disp"<<endl;
	}
};
class sub1:public base
{
public:
	void disp()
	{
		cout<<"in sub1 disp"<<endl;
	}
};
class sub2:public base
{
public:
	void disp()
	{
		cout<<"in sub2 disp"<<endl;
	}
};
void show(base *b)
{
	sub2 *ptr=dynamic_cast<sub2*>(b);
	if(ptr)
	{
		ptr->disp();
	}
}
int main()
{
	show(new sub1);
	show(new sub2);
}
